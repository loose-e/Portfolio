This is my solution to the first problem on projecteuler.net, found here:

https://projecteuler.net/problem=1

let aNum = []; for (i = 1; i < 1000; i++ ) { aNum.push(false); } for (i = 3; i < 1000; i+=3) { aNum[i] = true; } for (i = 5; i < 1000; i+=5) { aNum[i] = true; } let totalSum = 0; for (number in aNum) { if (aNum[number]) { totalSum += Number(number); } } console.log(totalSum);
